\begin{frame}{Prerequisites}\Large
	\begin{itemize}
		\item Group set up with two members (server and client).
			\bigskip

			Client a bit special (next slide)
			\bigskip

		\item AEAD algorithm is deterministic (Flag in COSE registry).
			\bigskip

		\item The key is only ever used to encrypt this very message.
			\bigskip

			KID details goes in there, no variability in AEAD left.
			\bigskip

		\item All clients generate same request (“Accept”?)
			\bigskip

			Out of scope.
	\end{itemize}
\end{frame}
